# Shared dependencies for both PCs.
# The operator PC needs all of them: pip install -r requirements.txt
#
# The ROBOT PC only strictly needs eclipse-zenoh, dmcan_sdk and pyusb
# (+ opencv-python if it has no working system OpenCV already). DO NOT
# blindly `pip install -r requirements.txt` for a venv on the robot: the
# generic opencv-python PyPI wheel is built WITHOUT GStreamer and its V4L2
# backend has failed outright ("can't open camera by index") on at least one
# UVC camera we use, while the apt-installed system OpenCV (GStreamer-enabled)
# opens the same camera fine. On the robot, create the venv with
# --system-site-packages and only install what's actually missing so it
# inherits the working system cv2 instead of shadowing it with a broken one:
#     python3 -m venv --system-site-packages .venv
#     .venv/bin/pip install eclipse-zenoh dmcan_sdk pyusb
#
# robot/damiao.py and robot/dlls/libdm_device.so are vendored in this repo
# (not on PyPI): a modified copy of the DAMIAO vendor SDK with the DMS2325
# motor type added, plus its native libusb backend. See robot/robot_agent.py.

dmcan_sdk>=1.0.1

fastapi>=0.110.0
uvicorn>=0.29.0
